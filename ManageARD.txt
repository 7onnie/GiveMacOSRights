-- Admin-Credentials: benutzt wenn das User-Feld im Auth-Dialog leer/fokussiert
-- ist (eingeloggter User ist KEIN Admin -> Admin muss eingegeben werden).
property _AdminUser : "praxis"
property _AdminPass : "praxis"

-- Fallback-/User-Credentials (optional): benutzt wenn im Auth-Dialog bereits
-- dieser Username vorausgefüllt ist, weil er AKTUELL Admin ist und macOS sein
-- Passwort verlangt (User-Feld dann nicht änderbar). "" = Fallback aus.
property _FallbackUser : "praxis"
property _FallbackPass : "praxis"

-- "Activate"   -> Ersteinrichtung (alles einschalten inkl. Sub-Optionen)
-- "Reactivate" -> Alle Toggles & Sub-Optionen einmal aus und wieder an
property _Mode : "Activate"

property _DelayClick : 0.1      -- Warten nach Klick
property _DelayAuth : 0.5       -- Warten nach Auth-Dialog
property _DelayNav : 0.25        -- Warten nach Pane-Wechsel
property _DelayPanel : 0.125      -- Warten nach Info-Panel/Sheet öffnen
property _SecAgentDone : false   -- Nach erstem Check kein SecurityAgent mehr prüfen


-- #########################################################
-- #  HANDLER: Auth-Felder zuverlässig per keystroke füllen #
-- #  Passwort-Feld per subrole (AXSecureTextField) finden, #
-- #  optionales Username-Feld = nicht-secure text field.   #
-- #  Beide per keystroke; Username mit Tab committen.       #
-- #  WICHTIG: "set value" committet den Username NICHT      #
-- #  (springt beim Submit zurück) und wird auf Secure-      #
-- #  Feldern ignoriert (Passwort bliebe leer) - macOS 26.   #
-- #########################################################

on fillAuthByKeystroke(_U, _P, _Container)
    tell application "System Events"
        set _PassF to missing value
        set _UserF to missing value
        -- entire contents: Felder in JEDER Verschachtelungstiefe finden
        -- (im Sheet liegen Username/Passwort teils tiefer als direkte Kinder).
        -- Suchfeld (AXSearchField) ausschließen, sonst wird es als User erkannt.
        set _Els to {}
        try
            with timeout of 6 seconds
                set _Els to entire contents of _Container
            end timeout
        end try
        repeat with _E in _Els
            try
                if (class of _E) is text field then
                    set _SR to ""
                    try
                        set _SR to (subrole of _E) as text
                    end try
                    if _SR is "AXSecureTextField" then
                        if _PassF is missing value then set _PassF to contents of _E
                    else if _SR is not "AXSearchField" then
                        if _UserF is missing value then set _UserF to contents of _E
                    end if
                end if
            end try
        end repeat
        if _PassF is missing value then return false

        -- Vorausgefüllten Usernamen lesen. Steht dort ein Name, will macOS
        -- das Passwort DIESES Users (er ist aktuell Admin); das User-Feld ist
        -- dann NICHT beschreibbar (Fokus klebt am Passwort, kein Tab/Klick hin).
        set _PreUser to ""
        if _UserF is not missing value then
            try
                set _PreUser to (value of _UserF) as text
            end try
        end if

        -- FALL A: Fallback-User ist vorausgefüllt -> nur dessen Passwort tippen,
        -- Username NICHT anfassen. Passwortfeld ist bereits fokussiert.
        if _FallbackUser is not "" and _PreUser is _FallbackUser then
            try
                set focused of _PassF to true
            end try
            try
                delay 0.05
                keystroke "a" using {command down}
                delay 0.05
                keystroke _FallbackPass
                delay 0.1
            end try
            return true
        end if

        -- FALL B: Admin-Credentials. Username nur setzen wenn das Feld
        -- fokussierbar ist (SecurityAgent) bzw. schon fokussiert (leeres
        -- User-Feld). Im System-Settings-Sheet lässt sich der Fokus NICHT
        -- ins User-Feld bewegen - dann bleibt es beim reinen Passwort-Tippen.
        if _UserF is not missing value then
            set _UserFocused to false
            try
                set _UserFocused to (focused of _UserF)
            end try
            if not _UserFocused then
                try
                    set focused of _UserF to true
                    delay 0.1
                    try
                        set _UserFocused to (focused of _UserF)
                    end try
                end try
            end if
            if _UserFocused then
                try
                    keystroke "a" using {command down}
                    delay 0.05
                    keystroke _U
                    delay 0.1
                    key code 48   -- Tab: End-Editing feuern -> Username committen
                    delay 0.15
                end try
            end if
        end if
        -- Passwort tippen (Passwortfeld ist per Default fokussiert;
        -- set focused hilft zusätzlich beim SecurityAgent-Dialog).
        try
            set focused of _PassF to true
        end try
        try
            delay 0.05
            keystroke "a" using {command down}
            delay 0.05
            keystroke _P
            delay 0.1
        end try
        return true
    end tell
end fillAuthByKeystroke


-- ###################################
-- #  HANDLER: Auth-Dialog behandeln #
-- ###################################

on handleAuthDialog(_U, _P)
    delay 0.3

    if not _SecAgentDone then
        set _SecAgentDone to true
        tell application "System Events"
            if exists process "SecurityAgent" then
                tell process "SecurityAgent"
                    try
                        repeat with _W in every window
                            if my fillAuthByKeystroke(_U, _P, _W) then
                                delay 0.2
                                key code 36
                                return true
                            else
                                repeat with _Btn in every button of _W
                                    set _BN to name of _Btn
                                    if _BN is not missing value then
                                        if _BN is not in {"Abbrechen", "Cancel", "Nicht erlauben", "Don't Allow", "Deny", "Ablehnen", "Beenden & erneut öffnen"} then
                                            click _Btn
                                            delay 0.3
                                            return true
                                        end if
                                    end if
                                end repeat
                            end if
                        end repeat
                    end try
                end tell
            end if
        end tell
    end if

    tell application "System Events"
        if exists process "System Settings" then
            tell process "System Settings"
                try
                    repeat 2 times
                        if (count of sheets of window 1) > 0 then
                            tell sheet 1 of window 1
                                if my fillAuthByKeystroke(_U, _P, it) then
                                    delay 0.2
                                    key code 36
                                    delay 0.5
                                else
                                    set _Clicked to false
                                    repeat with _Btn in every button
                                        set _BN to name of _Btn
                                        if _BN is not missing value then
                                            if _BN is not in {"Abbrechen", "Cancel", "Nicht erlauben", "Don't Allow", "Deny", "Ablehnen", "Beenden & erneut öffnen"} then
                                                click _Btn
                                                set _Clicked to true
                                                delay 0.3
                                                exit repeat
                                            end if
                                        end if
                                    end repeat
                                    if not _Clicked then
                                        key code 36
                                        delay 0.3
                                    end if
                                end if
                            end tell
                        else
                            exit repeat
                        end if
                    end repeat
                end try
            end tell
        end if
    end tell

    return false
end handleAuthDialog


-- #####################################################
-- #  HANDLER: Checkbox per AXId in SubSA setzen       #
-- #  Durchsucht alle Groups in der Sharing-ScrollArea #
-- #####################################################

on setMainToggle(_AXId, _TargetVal, _SubSA)
    tell application "System Events"
        tell process "System Settings"
            try
                repeat with _El in every UI element of _SubSA
                    if class of _El is group then
                        repeat with _K in every UI element of _El
                            if class of _K is checkbox then
                                try
                                    if (value of attribute "AXIdentifier" of _K) = _AXId then
                                        if value of _K is not _TargetVal then
                                            click _K
                                            delay _DelayClick
                                            if my handleAuthDialog(_AdminUser, _AdminPass) then
                                                delay _DelayAuth
                                            end if
                                        end if
                                        return true
                                    end if
                                end try
                            end if
                        end repeat
                    end if
                end repeat
            end try
        end tell
    end tell
    return false
end setMainToggle


-- ###############################################
-- #  HANDLER: Checkbox per AXId im Sheet setzen #
-- #  Durchsucht alle Groups in Sheet-ScrollArea #
-- ###############################################

on setSheetCheckbox(_AXId, _TargetVal)
    tell application "System Events"
        tell process "System Settings"
            try
                set _SheetSA to scroll area 1 of group 1 of sheet 1 of window 1
                repeat with _Grp in every group of _SheetSA
                    repeat with _CB in every checkbox of _Grp
                        try
                            if (value of attribute "AXIdentifier" of _CB) = _AXId then
                                if value of _CB is not _TargetVal then
                                    click _CB
                                    delay _DelayClick
                                    -- Sequoia: VNC-Passwort-Sheet abbrechen
                                    try
                                        if (count of sheets of sheet 1 of window 1) > 0 then
                                            key code 53
                                            delay 0.2
                                        end if
                                    end try
                                end if
                                return true
                            end if
                        end try
                    end repeat
                end repeat
            end try
        end tell
    end tell
    return false
end setSheetCheckbox


-- ##########################################
-- #  HANDLER: Info-Button in SubSA klicken #
-- ##########################################

on clickInfoButton(_AXId, _SubSA)
    tell application "System Events"
        tell process "System Settings"
            try
                repeat with _El in every UI element of _SubSA
                    if class of _El is group then
                        repeat with _K in every UI element of _El
                            if class of _K is button then
                                try
                                    if (value of attribute "AXIdentifier" of _K) = _AXId then
                                        click _K
                                        return true
                                    end if
                                end try
                            end if
                        end repeat
                    end if
                end repeat
            end try
        end tell
    end tell
    return false
end clickInfoButton


-- ############################################################
-- #  HANDLER: Sheet schließen (button 2 of group 1 of sheet) #
-- ############################################################

on closeSheet()
    tell application "System Events"
        tell process "System Settings"
            try
                click button 2 of group 1 of sheet 1 of window 1
                delay 0.5
            end try
        end tell
    end tell
end closeSheet


-- #############################################################
-- #  HANDLER: Optionen-Checkboxen setzen (Sheet bleibt offen) #
-- #  Pfad: group 1 of SA 1 of group 1 of Optionen-Sheet       #
-- #  "Berichte erstellen" wird IMMER auf AUS gezwungen        #
-- #  (unabhängig von _TargetVal) – kein AXId, Match per Name  #
-- #  Schließen erfolgt inline per key code 36                 #
-- #############################################################

on setCheckboxesInOptionen(_TargetVal)
    tell application "System Events"
        tell process "System Settings"
            try
                -- Optionen-Sheet bestimmen (sheet 2 oder sheet 1 of sheet 1)
                set _Sh to missing value
                if (count of sheets of window 1) > 1 then
                    set _Sh to sheet 2 of window 1
                else if (count of sheets of sheet 1 of window 1) > 0 then
                    set _Sh to sheet 1 of sheet 1 of window 1
                end if

                if _Sh is not missing value then
                    -- Checkboxen: bekannter Pfad group 1 of SA 1 of group 1
                    set _CBArea to group 1 of scroll area 1 of group 1 of _Sh
                    repeat with _CB in every checkbox of _CBArea
                        -- SOLL pro Checkbox: "Berichte erstellen" bleibt AUS
                        set _Want to _TargetVal
                        try
                            set _CBName to name of _CB
                            if _CBName is not missing value and _CBName contains "Bericht" then
                                set _Want to 0
                            end if
                        end try
                        if value of _CB is not _Want then
                            click _CB
                            delay _DelayClick
                        end if
                    end repeat
                end if
            end try
        end tell
    end tell
end setCheckboxesInOptionen


-- ###############################
-- #  HAUPTPROGRAMM              #
-- ###############################

-- Sharing-Pane öffnen
tell application "System Settings"
    activate
    delay 1.0
end tell
do shell script "open 'x-apple.systempreferences:com.apple.preferences.sharing'"
delay _DelayNav

-- Referenz auf Sharing-Content (Sonoma: group 3, Sequoia: group 2, Fallback: flach)
set _SubSA to missing value
tell application "System Events"
    tell process "System Settings"
        try
            set _SubSA to scroll area 1 of group 1 of group 3 of splitter group 1 of group 1 of window 1
        end try
        if _SubSA is missing value then
            try
                set _SubSA to scroll area 1 of group 1 of group 2 of splitter group 1 of group 1 of window 1
            end try
        end if
        if _SubSA is missing value then
            try
                set _SubSA to scroll area 1 of group 1 of window 1
            end try
        end if
    end tell
end tell
if _SubSA is missing value then
    display dialog "ManageARD: Sharing-UI nicht gefunden – macOS-Version nicht unterstützt." buttons {"OK"} default button 1
    error number -128
end if

-- ## 1. Reactivate: erst alles deaktivieren ###############
if _Mode is "Reactivate" then
    setMainToggle("Skriptfernsteuerung für AppsToggle", 0, _SubSA)
    setMainToggle("Entfernte AnmeldungToggle", 0, _SubSA)
    setMainToggle("Entfernte VerwaltungToggle", 0, _SubSA)
    delay 1.0
end if

-- ## 2. Haupt-Toggles aktivieren ##########################
setMainToggle("Entfernte VerwaltungToggle", 1, _SubSA)
setMainToggle("Entfernte AnmeldungToggle", 1, _SubSA)
setMainToggle("Skriptfernsteuerung für AppsToggle", 1, _SubSA)

-- ## 3. Entfernte Verwaltung – Info-Panel #################
clickInfoButton("Entfernte VerwaltungToggle.infoButton", _SubSA)
delay _DelayPanel

-- Reactivate: Sub-Checkboxen zuerst auf AN setzen (cycle)
-- VNCControlScreenToggle ausgenommen: aktivieren öffnet Passwort-Feld inline
-- und verändert die UI-Struktur → nachfolgende Checkboxen nicht mehr findbar (Sequoia)
if _Mode is "Reactivate" then
    setSheetCheckbox("AlwaysShowInMenuToggle", 1)
    setSheetCheckbox("PublicScreenControlToggle", 1)
end if

-- SOLL-Werte setzen (beide Modi)
setSheetCheckbox("AlwaysShowInMenuToggle", 0)      -- Status in Menüleiste: AUS
setSheetCheckbox("PublicScreenControlToggle", 0)   -- Genehmigung anfordern: AUS
setSheetCheckbox("VNCControlScreenToggle", 0)      -- VNC mit Passwort: AUS
-- "Berichte erstellen" liegt NICHT hier, sondern als Zugriffsrecht
-- im Optionen-Sheet → wird in setCheckboxesInOptionen auf AUS gezwungen

-- Optionen einmal öffnen
tell application "System Events"
    tell process "System Settings"
        try
            click button 1 of scroll area 1 of group 1 of sheet 1 of window 1
        end try
    end tell
end tell
delay _DelayPanel

-- Reactivate: erst deaktivieren, dann aktivieren (cycle, Sheet bleibt offen)
if _Mode is "Reactivate" then
    setCheckboxesInOptionen(0)
end if
setCheckboxesInOptionen(1)

-- Optionen-Sheet schließen (Buttons haben name=missing value -> key code 36)
tell application "System Events"
    tell process "System Settings"
        key code 36
        delay 0.5
    end tell
end tell

-- Verwaltung-Sheet schließen
closeSheet()
delay 0.5

-- ## 4. Entfernte Anmeldung – Info-Panel ##################
-- Einmal öffnen; Reactivate: FDA erst deaktivieren, dann aktivieren
clickInfoButton("Entfernte AnmeldungToggle.infoButton", _SubSA)
delay _DelayPanel
tell application "System Events"
    tell process "System Settings"
        try
            set _FDACb to checkbox 1 of group 2 of scroll area 1 of group 1 of sheet 1 of window 1
            if _Mode is "Reactivate" then
                -- Cycle: erst deaktivieren wenn aktiv
                if value of _FDACb is 1 then
                    click _FDACb
                    delay _DelayClick
                    if my handleAuthDialog(_AdminUser, _AdminPass) then
                        delay _DelayAuth
                    end if
                end if
            end if
            -- Aktivieren wenn deaktiviert (beide Modi)
            if value of _FDACb is 0 then
                click _FDACb
                delay _DelayClick
                my handleAuthDialog(_AdminUser, _AdminPass)
                delay _DelayAuth
            end if
        end try
    end tell
end tell
closeSheet()

display notification "ManageARD " & _Mode & " abgeschlossen." with title "ManageARD"

tell application "System Settings" to quit
