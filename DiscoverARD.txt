-- ===========================================================
-- DiscoverARD.applescript
-- Kompaktes LOG: ManageARD-relevante Werte prüfen
-- ===========================================================

tell application "System Settings"
    activate
    delay 1.0
end tell
do shell script "open 'x-apple.systempreferences:com.apple.preferences.sharing'"
delay 1.5

set _Log to "=== DiscoverARD ===" & return & return

tell application "System Events"
    tell process "System Settings"
        set _SubSA to scroll area 1 of group 1 of group 3 of splitter group 1 of group 1 of window 1

        -- ── 1. Haupt-Toggles ──────────────────────────────────
        set _ToggleIds to {"Entfernte VerwaltungToggle", "Entfernte AnmeldungToggle", "Skriptfernsteuerung für AppsToggle"}
        repeat with _TId in _ToggleIds
            set _Found to false
            repeat with _El in every UI element of _SubSA
                if _Found then exit repeat
                if class of _El is group then
                    repeat with _K in every UI element of _El
                        if class of _K is checkbox then
                            try
                                if (value of attribute "AXIdentifier" of _K) = _TId then
                                    set _Log to _Log & _TId & ": " & (value of _K) & return
                                    set _Found to true
                                    exit repeat
                                end if
                            end try
                        end if
                    end repeat
                end if
            end repeat
        end repeat

        -- ── 2. Verwaltung Info-Panel ──────────────────────────
        set _Found to false
        repeat with _El in every UI element of _SubSA
            if _Found then exit repeat
            if class of _El is group then
                repeat with _K in every UI element of _El
                    if class of _K is button then
                        try
                            if (value of attribute "AXIdentifier" of _K) = "Entfernte VerwaltungToggle.infoButton" then
                                click _K
                                set _Found to true
                                exit repeat
                            end if
                        end try
                    end if
                end repeat
            end if
        end repeat
        delay 1.0

        set _Log to _Log & return & "--- Verwaltung Sheet ---" & return

        -- Sheet-Checkboxen (AlwaysShowInMenu etc.)
        try
            set _SA1 to scroll area 1 of group 1 of sheet 1 of window 1
            set _SubCBIds to {"AlwaysShowInMenuToggle", "PublicScreenControlToggle", "VNCControlScreenToggle"}
            repeat with _CId in _SubCBIds
                repeat with _Grp in every group of _SA1
                    repeat with _CB in every checkbox of _Grp
                        try
                            if (value of attribute "AXIdentifier" of _CB) = _CId then
                                set _Log to _Log & _CId & ": " & (value of _CB) & return
                            end if
                        end try
                    end repeat
                end repeat
            end repeat
        end try

        -- ── 3. Optionen-Sheet ────────────────────────────────
        try
            click button 1 of scroll area 1 of group 1 of sheet 1 of window 1
        end try
        delay 1.0

        set _Sh to missing value
        set _ShDesc to "nicht gefunden"
        if (count of sheets of window 1) > 1 then
            set _Sh to sheet 2 of window 1
            set _ShDesc to "sheet 2 of window 1"
        else if (count of sheets of sheet 1 of window 1) > 0 then
            set _Sh to sheet 1 of sheet 1 of window 1
            set _ShDesc to "sheet 1 of sheet 1 of window 1"
        end if

        set _Log to _Log & return & "--- Optionen (" & _ShDesc & ") ---" & return

        if _Sh is not missing value then
            -- Checkboxen über bekannten Pfad
            try
                set _CBArea to group 1 of scroll area 1 of group 1 of _Sh
                set _CBList to every checkbox of _CBArea
                set _Log to _Log & "CBs (Grp1>SA1>Grp1): " & (count _CBList) & return
                set _CBIdx to 0
                repeat with _CB in _CBList
                    set _CBIdx to _CBIdx + 1
                    set _Log to _Log & "  CB" & _CBIdx & "=" & (value of _CB) & return
                end repeat
            on error _Err
                set _Log to _Log & "Fehler bekannter Pfad: " & _Err & return
            end try

            -- Buttons (zum Schließen)
            try
                set _GBtns to every button of group 1 of _Sh
                set _Log to _Log & "Grp1-Buttons: " & (count _GBtns)
                repeat with _Btn in _GBtns
                    set _BN to "(missing value)"
                    try
                        if name of _Btn is not missing value then
                            set _BN to name of _Btn
                        end if
                    end try
                    set _Log to _Log & " [" & _BN & "]"
                end repeat
                set _Log to _Log & return
            end try
        end if

        -- Optionen schließen
        key code 36
        delay 0.5

        -- Verwaltung schließen
        try
            click button 2 of group 1 of sheet 1 of window 1
            delay 0.5
        end try

        -- ── 4. Anmeldung Info-Panel ───────────────────────────
        set _Found to false
        repeat with _El in every UI element of _SubSA
            if _Found then exit repeat
            if class of _El is group then
                repeat with _K in every UI element of _El
                    if class of _K is button then
                        try
                            if (value of attribute "AXIdentifier" of _K) = "Entfernte AnmeldungToggle.infoButton" then
                                click _K
                                set _Found to true
                                exit repeat
                            end if
                        end try
                    end if
                end repeat
            end if
        end repeat
        delay 1.0

        set _Log to _Log & return & "--- Anmeldung Sheet ---" & return

        try
            set _FDACb to checkbox 1 of group 2 of scroll area 1 of group 1 of sheet 1 of window 1
            set _Log to _Log & "FDA (group2 CB1): " & (value of _FDACb) & return
        on error _Err
            set _Log to _Log & "FDA Fehler: " & _Err & return
        end try

        -- Anmeldung schließen
        try
            click button 2 of group 1 of sheet 1 of window 1
        end try
    end tell
end tell

display dialog _Log
tell application "System Settings" to quit
