-- Admin-Credentials: benutzt wenn das User-Feld im Auth-Dialog leer/fokussiert
-- ist (eingeloggter User ist KEIN Admin -> Admin muss eingegeben werden).
property _AdminUser : "praxis"
property _AdminPass : "praxis"

-- Fallback-/User-Credentials (optional): benutzt wenn im Auth-Dialog bereits
-- dieser Username vorausgefüllt ist, weil er AKTUELL Admin ist und macOS sein
-- Passwort verlangt (User-Feld dann nicht änderbar). "" = Fallback aus.
property _FallbackUser : "praxis"
property _FallbackPass : "praxis"

-- Lockfile: GLEICHER Pfad wie in ManageARD -> beide Scripts warten gegenseitig
-- aufeinander (gleichzeitiger Start möglich, laufen nacheinander).
property _GlobalLockFile : "/tmp/tomedo_macrights.lock"
property _GlobalLockWaitTime : 120

property _DelayNav : 1.0       -- Warten nach Sektions-Klick (Ladezeit)
property _DelayClick : 0.1     -- Warten nach Checkbox-Klick
property _DelayAuth : 0.3      -- Warten nach Auth-Dialog

-- Ziel-Sektionen: exakte deutsche AXIdentifier ohne "_Navigator"-Suffix.
-- "Vollständiger Festplattenzugriff" ist NICHT in der Sidebar –
-- wird separat per URL-Scheme geöffnet (siehe Hauptprogramm).
property _TargetSections : {¬
    "App-Verwaltung", ¬
    "Aufnahme von Bildschirm & Systemaudio", ¬
    "Automation", ¬
    "Bedienungshilfen", ¬
    "Bluetooth", ¬
    "Eingabeüberwachung", ¬
    "Kamera", ¬
    "Lokales Netzwerk", ¬
    "Mikrofon", ¬
    "Remote Desktop", ¬
    "Spracherkennung", ¬
    "Vollständiger Festplattenzugriff" ¬
}


-- ╔══════════════════════════════════════════════════════════╗
-- ║  HANDLER: Auth-Felder zuverlässig per keystroke füllen  ║
-- ║  Passwort-Feld per subrole (AXSecureTextField) finden,  ║
-- ║  optionales Username-Feld = nicht-secure text field.    ║
-- ║  Beide per keystroke; Username mit Tab committen.       ║
-- ║  WICHTIG: "set value" committet den Username NICHT      ║
-- ║  (springt beim Submit zurück) und wird auf Secure-      ║
-- ║  Feldern ignoriert (Passwort bliebe leer) - macOS 26.   ║
-- ╚══════════════════════════════════════════════════════════╝

on fillAuthByKeystroke(_U, _P, _Container)
    tell application "System Events"
        set _PassF to missing value
        set _UserF to missing value
        -- entire contents: Felder in JEDER Verschachtelungstiefe finden
        -- (im Sheet liegen Username/Passwort teils tiefer als direkte Kinder).
        -- Suchfeld (AXSearchField) ausschließen, sonst wird es als User erkannt.
        set _Els to {}
        try
            with timeout of 6 seconds
                set _Els to entire contents of _Container
            end timeout
        end try
        repeat with _E in _Els
            try
                if (class of _E) is text field then
                    set _SR to ""
                    try
                        set _SR to (subrole of _E) as text
                    end try
                    if _SR is "AXSecureTextField" then
                        if _PassF is missing value then set _PassF to contents of _E
                    else if _SR is not "AXSearchField" then
                        if _UserF is missing value then set _UserF to contents of _E
                    end if
                end if
            end try
        end repeat
        if _PassF is missing value then return false

        -- Vorausgefüllten Usernamen lesen. Steht dort ein Name, will macOS
        -- das Passwort DIESES Users (er ist aktuell Admin); das User-Feld ist
        -- dann NICHT beschreibbar (Fokus klebt am Passwort, kein Tab/Klick hin).
        set _PreUser to ""
        if _UserF is not missing value then
            try
                set _PreUser to (value of _UserF) as text
            end try
        end if

        -- FALL A: Fallback-User ist vorausgefüllt -> nur dessen Passwort tippen,
        -- Username NICHT anfassen. Passwortfeld ist bereits fokussiert.
        if _FallbackUser is not "" and _PreUser is _FallbackUser then
            try
                set focused of _PassF to true
            end try
            try
                delay 0.05
                keystroke "a" using {command down}
                delay 0.05
                keystroke _FallbackPass
                delay 0.1
            end try
            return true
        end if

        -- FALL B: Admin-Credentials. Username nur setzen wenn das Feld
        -- fokussierbar ist (SecurityAgent) bzw. schon fokussiert (leeres
        -- User-Feld). Im System-Settings-Sheet lässt sich der Fokus NICHT
        -- ins User-Feld bewegen - dann bleibt es beim reinen Passwort-Tippen.
        if _UserF is not missing value then
            set _UserFocused to false
            try
                set _UserFocused to (focused of _UserF)
            end try
            if not _UserFocused then
                try
                    set focused of _UserF to true
                    delay 0.1
                    try
                        set _UserFocused to (focused of _UserF)
                    end try
                end try
            end if
            if _UserFocused then
                try
                    keystroke "a" using {command down}
                    delay 0.05
                    keystroke _U
                    delay 0.1
                    key code 48   -- Tab: End-Editing feuern -> Username committen
                    delay 0.15
                end try
            end if
        end if
        -- Passwort tippen (Passwortfeld ist per Default fokussiert;
        -- set focused hilft zusätzlich beim SecurityAgent-Dialog).
        try
            set focused of _PassF to true
        end try
        try
            delay 0.05
            keystroke "a" using {command down}
            delay 0.05
            keystroke _P
            delay 0.1
        end try
        return true
    end tell
end fillAuthByKeystroke


-- ╔══════════════════════════════════════════════════════════╗
-- ║  HANDLER: Auth- UND Bestätigungs-Dialoge behandeln      ║
-- ║  SecurityAgent: bei jeder Section prüfen (jede Section  ║
-- ║  kann einen eigenen Auth-Dialog auslösen, z.B. Remote   ║
-- ║  Desktop). System-Settings-Sheet: immer prüfen.         ║
-- ╚══════════════════════════════════════════════════════════╝

on handleAuthDialog(_U, _P)
    delay 0.3

    -- ── Fall 1: SecurityAgent (Schloss-Dialog) ──────────────
    -- Jede Section kann einen eigenen SecurityAgent-Dialog auslösen
    tell application "System Events"
        if exists process "SecurityAgent" then
            tell process "SecurityAgent"
                try
                    repeat with _W in every window
                        if my fillAuthByKeystroke(_U, _P, _W) then
                            delay 0.2
                            key code 36 -- Return
                            return true
                        else
                            repeat with _Btn in every button of _W
                                set _BN to name of _Btn
                                if _BN is not missing value then
                                    if _BN is not in {"Abbrechen", "Cancel", "Nicht erlauben", "Don't Allow", "Deny", "Ablehnen", "Beenden & erneut öffnen"} then
                                        click _Btn
                                        delay 0.3
                                        return true
                                    end if
                                end if
                            end repeat
                        end if
                    end repeat
                end try
            end tell
        end if
    end tell

    -- ── Fall 2: Sheet in System Settings ────────────────────
    tell application "System Events"
        if exists process "System Settings" then
            tell process "System Settings"
                try
                    repeat 2 times
                        if (count of sheets of window 1) > 0 then
                            tell sheet 1 of window 1
                                if my fillAuthByKeystroke(_U, _P, it) then
                                    delay 0.2
                                    key code 36 -- Return
                                    delay 0.5
                                else
                                    -- Bestätigungs-Sheet (kein Passwortfeld)
                                    -- "Beenden & erneut öffnen" überspringen → "Nein" klicken
                                    set _Clicked to false
                                    repeat with _Btn in every button
                                        set _BN to name of _Btn
                                        if _BN is not missing value then
                                            if _BN is not in {"Abbrechen", "Cancel", "Nicht erlauben", "Don't Allow", "Deny", "Ablehnen", "Beenden & erneut öffnen"} then
                                                click _Btn
                                                set _Clicked to true
                                                delay 0.3
                                                exit repeat
                                            end if
                                        end if
                                    end repeat
                                    if not _Clicked then
                                        key code 36 -- Return als Fallback
                                        delay 0.3
                                    end if
                                end if
                            end tell
                        else
                            exit repeat
                        end if
                    end repeat
                end try
            end tell
        end if
    end tell

    return false
end handleAuthDialog


-- ╔══════════════════════════════════════════════════════════╗
-- ║  HANDLER: Zurück zur Hauptansicht navigieren            ║
-- ╚══════════════════════════════════════════════════════════╝

on navigateBack(_U, _P)
    my handleAuthDialog(_U, _P)
    delay 0.3
    tell application "System Settings"
        try
            set the current pane to pane id "com.apple.settings.PrivacySecurity.extension"
        on error
            delay 1.0
            my handleAuthDialog(_U, _P)
            delay 0.5
            try
                set the current pane to pane id "com.apple.settings.PrivacySecurity.extension"
            end try
        end try
    end tell
    delay 0.8
end navigateBack


-- ╔══════════════════════════════════════════════════════════╗
-- ║  HANDLER: Navigator-Button in Sidebar klicken           ║
-- ║  Pass 1: sichtbare Buttons in group 4                   ║
-- ║  Pass 2: Sidebar nach unten scrollen, nochmals suchen   ║
-- ╚══════════════════════════════════════════════════════════╝

on clickNavSection(_SectionName)
    tell application "System Events"
        tell process "System Settings"
            try
                -- group 3 (Sonoma/Tahoe) oder group 2 (Sequoia) Fallback
                set _CG to missing value
                try
                    set _CG to group 3 of splitter group 1 of group 1 of window 1
                end try
                if _CG is missing value then
                    try
                        set _CG to group 2 of splitter group 1 of group 1 of window 1
                    end try
                end if
                if _CG is missing value then error "Content group nicht gefunden"

                set _NavGroup to group 4 of scroll area 1 of group 1 of _CG

                -- Pass 1: aktuell sichtbare Buttons
                repeat with _E in every UI element of _NavGroup
                    if class of _E is button then
                        try
                            if (value of attribute "AXIdentifier" of _E) = (_SectionName & "_Navigator") then
                                click _E
                                return true
                            end if
                        end try
                    end if
                end repeat

                -- Pass 2: Sidebar nach unten scrollen
                try
                    set _SA to scroll area 1 of group 1 of _CG
                    set value of scroll bar 1 of _SA to 1.0
                    delay 0.5
                    set _NavGroup2 to group 4 of scroll area 1 of group 1 of _CG
                    repeat with _E in every UI element of _NavGroup2
                        if class of _E is button then
                            try
                                if (value of attribute "AXIdentifier" of _E) = (_SectionName & "_Navigator") then
                                    click _E
                                    return true
                                end if
                            end try
                        end if
                    end repeat
                    set value of scroll bar 1 of _SA to 0.0
                end try

            end try
        end tell
    end tell
    return false
end clickNavSection


-- ╔══════════════════════════════════════════════════════════╗
-- ║  HANDLER: Alle Checkboxen in aktueller Sektion          ║
-- ╚══════════════════════════════════════════════════════════╝

on enableCurrentSection(_U, _P)
    set _Enabled to 0
    set _NeedsRetry to true
    set _SkipCount to 0

    -- Pre-Pass: eingeklappte Disclosure-Rows aufklappen (z.B. Automation: tomedo → Finder)
    -- Stoppt wenn sich die Zahl der Outline-Elemente nicht mehr erhöht (Leaf-Rows ignoriert).
    tell application "System Events"
        tell process "System Settings"
            set _CG to missing value
            try
                set _CG to group 3 of splitter group 1 of group 1 of window 1
            end try
            if _CG is missing value then
                try
                    set _CG to group 2 of splitter group 1 of group 1 of window 1
                end try
            end if
            set _PrevCount to -1
            set _KeepExpanding to true
            repeat while _KeepExpanding
                set _KeepExpanding to false
                try
                    set _Outline to outline 1 of scroll area 1 of group 1 of scroll area 1 of group 1 of _CG
                    set _CurrCount to count of every UI element of _Outline
                    if _CurrCount > _PrevCount then
                        set _PrevCount to _CurrCount
                        repeat with _Row in every UI element of _Outline
                            if class of _Row is row then
                                try
                                    if value of attribute "AXDisclosing" of _Row is false then
                                        set value of attribute "AXDisclosing" of _Row to true
                                        set _KeepExpanding to true
                                        delay 0.1
                                    end if
                                end try
                            end if
                        end repeat
                    end if
                end try
            end repeat
        end tell
    end tell

    repeat while _NeedsRetry
        set _NeedsRetry to false
        set _FoundIdx to 0

        tell application "System Events"
            tell process "System Settings"
                try
                    set _CG to missing value
                    try
                        set _CG to group 3 of splitter group 1 of group 1 of window 1
                    end try
                    if _CG is missing value then
                        try
                            set _CG to group 2 of splitter group 1 of group 1 of window 1
                        end try
                    end if
                    if _CG is missing value then error "Content group nicht gefunden"
                    set _Outline to outline 1 of scroll area 1 of group 1 of scroll area 1 of group 1 of _CG
                    repeat with _Row in every UI element of _Outline
                        if class of _Row is row then
                            set _RowDone to false

                            repeat with _C1 in every UI element of _Row
                                if _RowDone then exit repeat

                                -- Tiefe 1: Checkbox direkt in Row
                                if class of _C1 is checkbox then
                                    if _FoundIdx < _SkipCount then
                                        set _FoundIdx to _FoundIdx + 1
                                    else
                                        if value of _C1 is 0 then
                                            click _C1
                                            delay _DelayClick
                                            if value of _C1 is 0 then
                                                if my handleAuthDialog(_U, _P) then
                                                    delay _DelayAuth
                                                end if
                                                if value of _C1 is 0 then
                                                    set _NeedsRetry to true
                                                    set _SkipCount to _SkipCount + 1
                                                    set _RowDone to true
                                                    exit repeat
                                                end if
                                            end if
                                            set _Enabled to _Enabled + 1
                                        end if
                                        set _RowDone to true
                                        exit repeat
                                    end if

                                else if class of _C1 is UI element then
                                    -- Tiefe 2 + 3
                                    repeat with _C2 in every UI element of _C1
                                        if _RowDone then exit repeat

                                        if class of _C2 is checkbox then
                                            if _FoundIdx < _SkipCount then
                                                set _FoundIdx to _FoundIdx + 1
                                            else
                                                if value of _C2 is 0 then
                                                    click _C2
                                                    delay _DelayClick
                                                    if value of _C2 is 0 then
                                                        if my handleAuthDialog(_U, _P) then
                                                            delay _DelayAuth
                                                        end if
                                                        if value of _C2 is 0 then
                                                            set _NeedsRetry to true
                                                            set _SkipCount to _SkipCount + 1
                                                            set _RowDone to true
                                                            exit repeat
                                                        end if
                                                    end if
                                                    set _Enabled to _Enabled + 1
                                                end if
                                                set _RowDone to true
                                                exit repeat
                                            end if

                                        else if class of _C2 is UI element then
                                            -- Tiefe 3
                                            repeat with _C3 in every UI element of _C2
                                                if _RowDone then exit repeat
                                                if class of _C3 is checkbox then
                                                    if _FoundIdx < _SkipCount then
                                                        set _FoundIdx to _FoundIdx + 1
                                                    else
                                                        if value of _C3 is 0 then
                                                            click _C3
                                                            delay _DelayClick
                                                            if value of _C3 is 0 then
                                                                if my handleAuthDialog(_U, _P) then
                                                                    delay _DelayAuth
                                                                end if
                                                                if value of _C3 is 0 then
                                                                    set _NeedsRetry to true
                                                                    set _SkipCount to _SkipCount + 1
                                                                    set _RowDone to true
                                                                    exit repeat
                                                                end if
                                                            end if
                                                            set _Enabled to _Enabled + 1
                                                        end if
                                                        set _RowDone to true
                                                        exit repeat
                                                    end if
                                                end if
                                            end repeat -- C3
                                        end if
                                        if _RowDone then exit repeat
                                    end repeat -- C2
                                end if
                                if _RowDone then exit repeat
                            end repeat -- C1
                            if _NeedsRetry then exit repeat
                        end if
                    end repeat -- Row
                end try
            end tell
        end tell
    end repeat

    -- Fallback: Flat-Group-Struktur (macOS 26: Remote Desktop hat kein outline)
    -- SA1 enthält direkt eine group mit Checkboxen statt einer verschachtelten SA+outline
    tell application "System Events"
        tell process "System Settings"
            try
                set _CG to missing value
                try
                    set _CG to group 3 of splitter group 1 of group 1 of window 1
                end try
                if _CG is missing value then
                    try
                        set _CG to group 2 of splitter group 1 of group 1 of window 1
                    end try
                end if
                if _CG is missing value then error "Content group nicht gefunden"
                set _SA1 to scroll area 1 of group 1 of _CG
                repeat with _FEl in every UI element of _SA1
                    if class of _FEl is group then
                        repeat with _FC in every UI element of _FEl
                            if class of _FC is checkbox then
                                try
                                    if value of _FC is 0 then
                                        click _FC
                                        delay _DelayClick
                                        if value of _FC is 0 then
                                            if my handleAuthDialog(_U, _P) then
                                                delay _DelayAuth
                                            end if
                                        end if
                                        if value of _FC is 1 then
                                            set _Enabled to _Enabled + 1
                                        end if
                                    end if
                                end try
                            end if
                        end repeat
                    end if
                end repeat
            end try
        end tell
    end tell

    return _Enabled
end enableCurrentSection


-- ╔══════════════════════════════════════════════════════════╗
-- ║  HANDLER: Lockfile (geteilt mit ManageARD)               ║
-- ║  ATOMAR per mkdir (Lock = Verzeichnis) -> auch bei exakt ║
-- ║  gleichzeitigem Start gewinnt nur einer (touch/-f hätte  ║
-- ║  eine Race Condition). LOCK wartet bis _MaxWaitSeconds,  ║
-- ║  danach Übernahme (Zero-Touch). Stale >300s = Crash ->   ║
-- ║  sofort übernehmen.                                      ║
-- ╚══════════════════════════════════════════════════════════╝

on ManageLockFile(_Action, _LockFilePath, _MaxWaitSeconds)
    if _Action is "LOCK" then
        -- Kompletter Warte-/Stale-Check in EINEM Shell-Aufruf.
        -- Status: free/waited/stale/stolen. Lock am Ende immer gesetzt (touch).
        set _MaxIter to _MaxWaitSeconds * 2
        set _LockStatus to "free"
        try
            set _LockStatus to do shell script "LOCK=" & quoted form of _LockFilePath & "; STALE=300; st=free; i=0; while ! mkdir \"$LOCK\" 2>/dev/null; do age=$(( $(date +%s) - $(stat -f %m \"$LOCK\" 2>/dev/null || echo 0) )); if [ $age -gt $STALE ]; then rm -rf \"$LOCK\"; st=stale; continue; fi; if [ $i -ge " & _MaxIter & " ]; then rm -rf \"$LOCK\"; st=stolen; continue; fi; st=waited; sleep 0.5; i=$((i+1)); done; echo $st"
        on error _LockErr
            log "ManageLockFile: Shell-Fehler (" & _LockErr & ") – setze Lock direkt."
            try
                do shell script "mkdir -p " & quoted form of _LockFilePath
            end try
        end try

        if _LockStatus is "stale" then
            log "Lockfile war stale (>300s, vermutlich Crash-Überbleibsel) – übernommen."
        else if _LockStatus is "stolen" then
            log "WARNUNG: Lock nach " & _MaxWaitSeconds & "s nicht freigegeben – übernommen (Zero-Touch hat Vorrang, paralleler Lauf möglich)."
        else if _LockStatus is "waited" then
            log "Lock wurde freigegeben – übernommen."
        end if

    else if _Action is "UNLOCK" then
        try
            do shell script "rm -rf " & quoted form of _LockFilePath
        end try
    end if
end ManageLockFile


-- ╔══════════════════════════════════════════════════════════╗
-- ║  HAUPTPROGRAMM                                          ║
-- ╚══════════════════════════════════════════════════════════╝

-- Lock setzen (wartet ggf. auf ManageARD)
my ManageLockFile("LOCK", _GlobalLockFile, _GlobalLockWaitTime)

try

tell application "System Settings"
    activate
    delay 1.5
    set the current pane to pane id "com.apple.settings.PrivacySecurity.extension"
    delay 1.0
end tell

set _TotalEnabled to 0
set _SectionsProcessed to 0
set _SectionsSkipped to 0

repeat with _SectionName in _TargetSections
    -- contents of: gibt String-Wert statt Loop-Referenz → Vergleiche funktionieren korrekt
    set _S to contents of _SectionName

    if _S is "Vollständiger Festplattenzugriff" then
        -- Nicht in Sidebar → direkt per URL-Scheme öffnen
        do shell script "open 'x-apple.systempreferences:com.apple.settings.PrivacySecurity.extension?Privacy_AllFiles'"
        delay 2.0
        set _SectionsProcessed to _SectionsProcessed + 1
        set _N to enableCurrentSection(_AdminUser, _AdminPass)
        set _TotalEnabled to _TotalEnabled + _N
        navigateBack(_AdminUser, _AdminPass)
    else
        set _Found to clickNavSection(_S)
        if _Found then
            set _SectionsProcessed to _SectionsProcessed + 1
            delay _DelayNav
            set _N to enableCurrentSection(_AdminUser, _AdminPass)
            set _TotalEnabled to _TotalEnabled + _N
            navigateBack(_AdminUser, _AdminPass)
        else
            set _SectionsSkipped to _SectionsSkipped + 1
        end if
    end if
end repeat

display notification ¬
    ((_SectionsProcessed as string) & " Sektionen, " & (_TotalEnabled as string) & " Rechte aktiviert, " & (_SectionsSkipped as string) & " nicht gefunden.") ¬
    with title "Privacy Setup abgeschlossen"

tell application "System Settings" to quit

-- Lock freigeben (normales Ende)
my ManageLockFile("UNLOCK", _GlobalLockFile, _GlobalLockWaitTime)

on error _MainErr number _MainNum
    -- Lock immer freigeben; unerwartete Fehler danach sichtbar weiterreichen
    -- (-128 = sauberer Abbruch -> schlucken).
    my ManageLockFile("UNLOCK", _GlobalLockFile, _GlobalLockWaitTime)
    if _MainNum is not -128 then error _MainErr number _MainNum
end try
